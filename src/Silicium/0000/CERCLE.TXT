
10 CLS
20 CX=60:CY=15:R=15
30 PSET(CX+R,CY)
40 PSET(CX,CY-R)
50 PSET(CX-R,CY)
60 PSET(CX,CY+R)
70 X=R
80 Y=1
90 S=1-X
100 GOTO 140
110 GOSUB 200
120 Y=Y+1
130 S=S+2*Y-1
140 IF S<0 THEN 110
150 X=X-1
160 S=S-2*X
170 IF X>Y THEN 110
180 GOSUB 200
185 CLS:CIRCLE (CX,CY),R
190 END
200 PSET(CX+X,CY-Y):PSET(CX-Y,CY+X)
210 PSET(CX+X,CY+Y):PSET(CX+Y,CY+X)
220 PSET(CX-X,CY-Y):PSET(CX+Y,CY-X)
230 PSET(CX-X,CY+Y):PSET(CX-Y,CY-X)
240 RETURN
