30 DIM X(N)
40 FOR J=1 TO N
50 INPUT "NOMBRE" ;X(J)
60 NEXT J
70 GOSUB 100
80 GOSUB 200
90 END
100 REM MAX
110 MAX=0
120 FOR K=1 TO N
130 IF MAX<X(K) THEN MAX=X(K)
140 NEXT K
150 PRINT "VALEUR MAX" ;MAX
158 GOTO 160
160 RETURN
200 REM MIN
210 MIN=100
220 FOR K=1 TO N
230 IF MIN>X(K) THEN MIN=X(K)
240 NEXT K
250 PRINT "VALEUR MIN" ;MIN
260 RETURN
